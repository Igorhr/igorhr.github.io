function triangleArea(a,b,c) {
var perimetar = (a+b+c)/2;
var area = Math.sqrt(perimetar*((perimetar-a)*(perimetar-b)*(perimetar-c)));
return area;
}

function evSmetaj(){
var a1 = document.querySelector("input[name=a]").value;
var b1 = document.querySelector("input[name=b]").value;
var c1 = document.querySelector("input[name=c]").value;
var ekran = document.querySelector("#ekran");
