JavaScript program to calculate days left until next Christmas.





CODE:

function daysUndtilNewYear() {
var ekran = document.querySelector("#ekran");
var today = new Date();
var godina = today.getFullYear()+1;
var newYear = new Date(godina, 0, 1);
var denovi = newYear - today;
var daysUntilDate = 1*denovi / (1000*60*60*24);
var roundDays = Math.round(daysUntilDate);
ekran.innerHTML = "there is " + roundDays + " days until New YEAR";
}
