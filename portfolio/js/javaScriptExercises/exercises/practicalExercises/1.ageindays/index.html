Calculate your age in days



function takeName(ime) {
ime = document.querySelector("#input1").value;
return ime;
}

function takeAge(godini) {
godini = document.querySelector("#input2").value;
return 1*godini*365;
}

function displayAgeInDays() {
var ekran = document.querySelector("#ekran");
var name = takeName();
var age = takeAge();
if(name && age) {
ekran.innerHTML = name + " your age in days is " + age + " days.";
ekran.style.display = "block";
} else {
alert("please enter your name and age");
ekran.style.display = "none";
}
}
